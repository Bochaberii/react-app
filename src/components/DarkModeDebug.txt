import React from 'react';
import { useTheme } from '../contexts/ThemeContext';

/**
 * Debug component to test dark mode functionality
 */
const DarkModeDebug = () => {
  const { isDark, toggleTheme } = useTheme();

  return (
    <div className="p-4 m-4 border-2 border-red-500">
      <h3 className="text-lg font-bold mb-2">Dark Mode Debug</h3>
      <p className="mb-2">Current theme: {isDark ? 'Dark' : 'Light'}</p>
      <p className="mb-2">HTML has dark class: {document.documentElement.classList.contains('dark') ? 'Yes' : 'No'}</p>
      <button 
        onClick={toggleTheme}
        className="px-4 py-2 bg-blue-500 text-white rounded hover:bg-blue-600"
      >
        Toggle Theme
      </button>
      <div className="mt-4 p-2 bg-gray-100 dark:bg-gray-800 text-gray-900 dark:text-white">
        This box should change color when theme toggles
      </div>
    </div>
  );
};

export default DarkModeDebug;
